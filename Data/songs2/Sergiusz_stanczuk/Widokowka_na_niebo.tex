% Preamble
\tytul{Widokówka na niebo}{sł. i muz. Sergiusz Stańczuk}{Sergiusz Stańczuk}
\begin{text}
    rozmowa niedokończona\\
    niedopite wino\\
    usta spalone słowem

    ogarki słów niedopalone\\
    w ustach jak papieros\\
    żarem kaleczą serce - heros

    czasem twarde jak kamień\\
    jak przerażony tłum\\
    zimne jak stal czyste jak łza

    czasem niczyje\\
    więc przygarniam je mocno do siebie\\
    wtedy mam niebo\\
    ciebie
\end{text}
